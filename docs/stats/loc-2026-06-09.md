## 复算

```bash
# 总行数(类型分布)
for ext in java py xml yml yaml sql sh md; do
  files=$(find . -type d \( -name target -o -name build -o -name logs -o -name .claude -o -name node_modules -o -name .idea -o -name .venv -o -name .git \) -prune -o -name "*.$ext" -type f -print 2>/dev/null | wc -l)
  lines=$(find . -type d \( -name target -o -name build -o -name logs -o -name .claude -o -name node_modules -o -name .idea -o -name .venv -o -name .git \) -prune -o -name "*.$ext" -type f -print 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  printf ".%-6s  %6s files  %8s lines\n" "$ext" "$files" "$lines"
done

# Java 按模块(main+test)
for m in batch-common batch-trigger batch-orchestrator batch-worker-core \
         batch-worker-import batch-worker-export batch-worker-process \
         batch-worker-dispatch batch-worker-atomic \
         batch-worker-sdk batch-worker-sdk-spring-boot-starter batch-worker-sdk-testkit \
         batch-console-api batch-e2e-tests; do
  main_lines=$(find "$m/src/main/java" -name '*.java' -type f 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  test_lines=$(find "$m/src/test/java" -name '*.java' -type f 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  echo "$m  main=$main_lines  test=$test_lines"
done

# Python SDK
find batch-worker-sdk-python/src -name '*.py' -type f | xargs cat | wc -l
find batch-worker-sdk-python/tests -name '*.py' -type f | xargs cat | wc -l
```
