## 复算

```bash
# 总行数(类型分布)
for ext in java xml yml sql sh md; do
  files=$(find . -type d \( -name target -o -name build -o -name logs -o -name .claude -o -name node_modules -o -name .idea \) -prune -o -name "*.$ext" -type f -print | wc -l)
  lines=$(find . -type d \( -name target -o -name build -o -name logs -o -name .claude -o -name node_modules -o -name .idea \) -prune -o -name "*.$ext" -type f -print | xargs cat | wc -l)
  printf ".%-5s  files=%s  lines=%s\n" "$ext" "$files" "$lines"
done

# Java 按模块
for m in batch-common batch-trigger batch-orchestrator batch-worker-{core,import,export,process,dispatch} batch-console-api; do
  main=$(find $m/src/main -name '*.java' 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  test=$(find $m/src/test -name '*.java' 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  printf "%-25s  main=%s  test=%s\n" "$m" "$main" "$test"
done
```
