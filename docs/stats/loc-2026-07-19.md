## 复算

```bash
# 总行数（类型分布）
for ext in java py xml yml yaml sql sh md; do
  tracked=$(git ls-files "*.$ext" | awk '!/(^|\/)(target|build|logs|\.claude|node_modules|\.idea|\.venv|\.git)(\/|$)/')
  files=$(printf '%s\n' "$tracked" | sed '/^$/d' | wc -l)
  lines=$(printf '%s\n' "$tracked" | xargs cat 2>/dev/null | wc -l)
  printf ".%-6s  %6s files  %8s lines\n" "$ext" "$files" "$lines"
done

# Java 按模块（main + test）
for m in batch-common batch-trigger batch-orchestrator \
         batch-worker/core batch-worker/import batch-worker/export \
         batch-worker/process batch-worker/dispatch batch-worker/atomic \
         sdk/java/core sdk/java/spring sdk/java/testkit \
         batch-console-api batch-e2e-tests; do
  main_lines=$(find "$m/src/main/java" -name '*.java' -type f 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  test_lines=$(find "$m/src/test/java" -name '*.java' -type f 2>/dev/null | xargs cat 2>/dev/null | wc -l)
  echo "$m  main=$main_lines  test=$test_lines"
done

# 多语言 SDK
find sdk/python/src -name '*.py' -type f | xargs cat | wc -l
find sdk/python/tests -name '*.py' -type f | xargs cat | wc -l
find sdk/go -name '*.go' -type f | xargs cat | wc -l
find sdk/rust -type d -name target -prune -o -name '*.rs' -type f -print | xargs cat | wc -l
find sdk/typescript -type d -name node_modules -prune -o -name '*.ts' -type f -print | xargs cat | wc -l
```
